CREATE PROCEDURE [dbo].[hsp_ViewAllMedecineStock]
AS
BEGIN
    SET NOCOUNT ON;

    SELECT
        MS.[medicineId],
        MS.[medecineName],
        MS.[medecineDescription],
        MS.[numberOfStock]
    FROM 
        [dbo].[MedecineStocks] MS
    WHERE 
        MS.[isActive] = 1 
    ORDER BY
        MS.[medecineName] ASC;
END
GO
