CREATE PROCEDURE [dbo].[hsp_ViewClinicStatus]
AS
BEGIN
    SET NOCOUNT ON;

    SELECT
        CS.[logId],
        CS.[status]
    FROM 
        [dbo].[ClinicStatus] CS
    WHERE 
        CS.[isActive] = 1 -- Only view active status logs
    ORDER BY
        CS.[logId] DESC; -- Usually you want the latest log first
END
GO
