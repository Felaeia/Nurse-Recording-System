-- b. nsp_ViewAppointmentScheduleList
-- Gets all active, non-deleted appointment schedules
CREATE PROCEDURE [dbo].[nsp_ViewAppointmentScheduleList]
AS
BEGIN
    SET NOCOUNT ON;

    SELECT
        [appointmentId],
        [appointmentTime],
        [appointmentDescription]
    FROM
        [dbo].[AppointmentSchedule]
    WHERE
        [deletedOn] IS NULL
        AND [isActive] = 1
    ORDER BY
        [appointmentTime] DESC;
END

GO
